Nested containers?

<main class="container">
  <header>...</header>
  <aside>...</aside>
  <section>
    <article class="container">
      <p>...</p>
    </article>
  </section>
</main>
 
main.container {
  container-name: main;
  /* Or together: */
  container: main / inline-size;
}
p { font-size: 150%; }
@container main (min-width: 40rem) {
  p { font-size: 200%; }
